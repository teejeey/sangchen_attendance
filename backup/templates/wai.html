function fetchStudents() { const cls = document.getElementById('className').value; const slot = document.getElementById('timeTable').value; const container = document.getElementById('takeAttendanceContainer'); const placeholder = document.getElementById('takePlaceholder'); const list = document.getElementById('studentList'); const saveBtn = document.getElementById('saveBtn'); if(!cls || !slot) { container.style.display = 'none'; placeholder.style.display = 'block'; saveBtn.disabled = true; return; } placeholder.style.display = 'none'; container.style.display = 'block'; saveBtn.disabled = true; list.innerHTML = '
'; fetch(`/get_students/${cls}?date=${document.getElementById('attDate').value}&subject=${encodeURIComponent(slot)}`) .then(res => res.json()).then(data => { const alreadyTaken = ( data.already_taken === true || data.already_taken === 'true' || data.already_taken === 'TRUE' ); if (alreadyTaken) { list.innerHTML = '
Already saved.'; } else if (data.students && data.students.length > 0) { let h = ''; data.students.forEach(s => { h += `${s.ID}${s.Name} `; }); list.innerHTML = h; saveBtn.disabled = false; } else { list.innerHTML = 'No students found.'; } }); }
